### LD based SNP pruning (--indep-pairwise 50 10 0.5) ###
plink --bfile HGTJAM_GSproject2_v3_auto_genoQC005hwe1e-18_diffmissQC_mindQC01 \
--indep-pairwise 50 10 0.1 \
--out HGTJAM_GSproject2_v3_auto_genoQC005hwe1e-18_diffmissQC_mindQC01_LDprune
# should be 50 10 0.1 - first value is window size (50 SNPs), second value is how many SNPs/”steps” you move when you move the window (10 SNPs), third value is the r^2. The larger the threshold, the less stringent, and the larger the window, the less stringent. 

plink --bfile HGTJAM_GSproject2_v3_auto_genoQC005hwe1e-18_diffmissQC_mindQC01 \
--extract HGTJAM_GSproject2_v3_auto_genoQC005hwe1e-18_diffmissQC_mindQC01_LDprune.prune.in \
--make-bed \
--out HGTJAM_GSproject2_v3_auto_genoQC005hwe1e-18_diffmissQC_mindQC01_pruned
wc -l HGTJAM_GSproject2_v3_auto_genoQC005hwe1e-18_diffmissQC_mindQC01_pruned.bim
# 278777 SNPs 

### Inbreeding coefficients ###
plink --bfile HGTJAM_GSproject2_v3_auto_genoQC005hwe1e-18_diffmissQC_mindQC01_pruned \
--het \
--out HGTJAM_GSproject2_v3_auto_genoQC005hwe1e-18_diffmissQC_mindQC01_pruned_het
# this automatically skips male X and Y thank goodness 
# so the F values range from -0.1 to 0.1. From reading the PLINK website, it seems like the farther away from 0 F is, the more it suggests inbreeding, and it may need to be filtered out. negative values are “more random than random” which is usually fine, but if it’s too negative then it may suggest contamination 

# go into R to make histogram of F values (inbreeding coefficient)
