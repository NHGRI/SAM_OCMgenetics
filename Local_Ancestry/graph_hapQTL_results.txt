let y denote phenotypes.
let g denotes SNP gentoype vector;
let a denote ancestry dosage (0-2);
a1(European) is first component
a2(African) is second component
bfc is log10 Bayes factor of haplotype association conditional on ancestral cluster dosage;
bfs is log10 BF of  y~a1+a2
bfs1 is log10BF of y~a1/2*g
bfs2 is log10BF of y~a2/2*g
bf1 is log10BF of y~genotype
pv1 is -log10 p-value of y~genotype
bf1i is log10BF of y~ imputed genotype dosages.
pv1i is -log10 p-value of y~ imputed genotype dosages.

}

read.table("HGTJAM_GSproject2_v3_auto_genoQC005hwe1e-18_diffmissQC_mindQC01_ibdQC.bim") -> bim

for(i in 1:22){

assign(paste("merged",i,sep=""), merged)
}

### here's where you start to pick bfc, bfs, etc 
for(i in 1:22){
